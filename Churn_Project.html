from sqlalchemy import create_engine
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

engine = create_engine("mysql+mysqlconnector://root:EMPP%40123swe08@localhost/churn_db")

query = """SELECT
  CASE
     WHEN Age < 30 THEN 'Under 30'
     WHEN Age BETWEEN 30 AND 50 THEN '30-50'
     ELSE 'Over 50'
  END AS age_group,
  COUNT(*) AS total,
  SUM(Exited) AS churned,
  ROUND(SUM(Exited)/COUNT(*)*100, 2) AS churn_rate
FROM customers GROUP BY age_group"""

df = pd.read_sql(query, engine)

sns.barplot(data=df, hue='age_group', x='churn_rate', palette='coolwarm', legend=True)
plt.title('Churn Rate by Age Group')
plt.show()

Interpretation - Customers over 50 show the highest churn rate, followed by those in the 30-50 range. Younger customers under 30 churn the least. Age is clearly a strong factor in customer retention.
